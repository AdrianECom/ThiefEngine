1 /**
  2 * @class
  3 * @classdesc Represents a 4-dimensional vector.
  4 * @param {Number} x The x component.
  5 * @param {Number} y The y component.
  6 * @param {Number} z The z component.
  7 * @param {Number} w The w component.
  8 */
  9 var Vector4 = function (x,y,z,w){
 10     this.x = x;
 11     this.y = y;
 12     this.z = z;
 13     this.w = w;
 14 };

 75 /**

 78 */
 79 Vector4.prototype.len = function (){
 80     var x = this.x;
 81     var y = this.y;
 82     var z = this.z;
 83 
 84     return Math.sqrt((x*x) + (y*y) + (z*z));

245 /**
246 * Returns the dot product between this vector and other vector.
247 * @param {Vector4} vec The other vector.
248 * @returns {Number} The dot product between this vector and other vector..
249 */
250 Vector4.prototype.dot = function (vec){
251   var x1 = this.x;
252   var y1 = this.y;
253   var z1 = this.z;
254 
255   var x2 = vec.x;
256   var y2 = vec.y;
257   var z2 = vec.z;
258 
259   return x1*x2 + y1*y2 + z1*z2;
260 };

264 /**
265 * Calculates the cross product between this vector and other vector.
266 * @param {Vector4} vec The other vector.
267 * @returns {Vector4} this.
268 */
269 Vector4.prototype.cross = function (vec){
270 
271   var x1 = this.x;
272   var y1 = this.y;
273   var z1 = this.z;
274 
275   var x2 = vec.x;
276   var y2 = vec.y;
277   var z2 = vec.z;
278 
279   this.x = y1 * z2 - z1 * y2;
280 	this.y = z1 * x2 - x1 * z2;
281 	this.z = x1 * y2 - y1 * x2;
282   this.w = 0;
283 
284   return this;
285 };

289 /**
290 * Returns the angle in degrees relative to other vector.
291 * @param {Vector4} vec The other vector.
292 * @returns {Number} The angle in degrees relative to other vector.
293 */
294 Vector4.prototype.ang = function (vec){
295     var dot = this.dot(vec);
296 
297     var len1 = this.len();
298     var len2 = vec.len();
299 
300     var cos = (dot/(len1*len2));
301 
302     if(len1*len2 === 0)
303       return 0;
304 
305     return Math.acos(cos);
306 };

313 */
314 Vector4.prototype.nor = function (){
315 
316     var len = this.len();
317 
318     if(len > 0){
319         this.x = this.x/len;
320         this.y = this.y/len;
321         this.z = this.z/len;
322         this.w = this.w/len;
323     }
324

330 /**
331 * Returns the distance between this vector and other vector.
332 * @param {Vector4} vec The other vector.
333 * @returns {Number} The distance between this vector and other vector.
334 */
335 Vector4.prototype.dst = function (vec){
336 
337     var deltaX = vec.x-this.x;
338     var deltaY = vec.y-this.y;
339     var deltaZ = vec.z-this.z;
340 
341     return Math.sqrt( (deltaX*deltaX) + (deltaY*deltaY) + (deltaZ*deltaZ) );
342 };

359 /**
360 * Tests if this vector is equal to the other vector, whit an epsilon error.
361 * @param {Vector4} vec The other vector.
362 * @param {Number} epsilon The error.
363 * @returns {boolean} this True if this vector is equal to the other vector, whit an epsilon error.
364 */
365 Vector4.prototype.epsilonEquals = function(vec,epsilon){
366     // Points are equal if each component is within 'epsilon' of each other
367     if( (Math.abs(this.x - vec.x) <= epsilon) &&
368         (Math.abs(this.y - vec.y) <= epsilon) &&
369         (Math.abs(this.z - vec.z) <= epsilon) )
370         return true;
371     else
372         return false;
373 };
