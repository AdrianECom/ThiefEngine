1 /**
  2 * @class
  3 * @classdesc This class manages multiple sprites wich have the same texture.
  4 */
  5 var SpriteBatch = function (material){
  6   this.material = material;
  7   this.renderers = {};
  8   this.renderContext = null;
  9   this.vboColor = 0;
 10 
 11 };
 12 
 13 SpriteBatch.mesh = new RectangleMesh();
 14 
 15 SpriteBatch.vboPosition = 0;
 16 SpriteBatch.vboElemIndices = 0;
 17 SpriteBatch.vboTexture = 0;
 18 SpriteBatch.vboNormal = 0;
 19 SpriteBatch.vao = 0;
 20 
 21 SpriteBatch.binded = false;

 25 /**
 26 * Adds a sprite renderer.
 27 * @param {SpriteRenderer} renderer The sprite renderer.
 28 */
 29 SpriteBatch.prototype.add = function (renderer){
 30 
 31   // stored by id
 32   if( ! (renderer.getId() in this.renderers))
 33     this.renderers[renderer.getId()] = renderer;
 34 };

 38 /**
 39 * Binds the batch.
 40 */
 41 SpriteBatch.prototype.bind = function (){
 42 
 43   this.material.bind();
 44 
 45   // only bind the rectangle once
 46   if(! SpriteBatch.binded){
 47 
 48     SpriteBatch.binded = true;
 49 
 50     // create VAO
 51     SpriteBatch.vao = vao_ext.createVertexArrayOES();
 52     vao_ext.bindVertexArrayOES(SpriteBatch.vao);
 53 
 54 
 55     // POSITION
 56     gl.enableVertexAttribArray(0);
 57     SpriteBatch.vboPosition = gl.createBuffer();
 58     gl.bindBuffer(gl.ARRAY_BUFFER, SpriteBatch.vboPosition);
 59     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(SpriteBatch.mesh.getVerticesData()), gl.STATIC_DRAW);
 60     gl.vertexAttribPointer(0, 4, gl.FLOAT, false, 0, 0);
 61 
 62 
 63     // TEXTURE
 64     gl.enableVertexAttribArray(1);
 65     SpriteBatch.vboTexture = gl.createBuffer();
 66     gl.bindBuffer(gl.ARRAY_BUFFER, SpriteBatch.vboTexture);
 67     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(SpriteBatch.mesh.getTexCoordData()), gl.STATIC_DRAW);
 68     gl.vertexAttribPointer(1, 2, gl.FLOAT, false, 0, 0);
 69 
 70 
 71 
 72 
 73 
 74     // NORMALS
 75     gl.enableVertexAttribArray(3);
 76     SpriteBatch.vboNormal = gl.createBuffer();
 77     gl.bindBuffer(gl.ARRAY_BUFFER, SpriteBatch.vboNormal);
 78     gl.bufferData(gl.ARRAY_BUFFER, new Float32Array( SpriteBatch.mesh.getNormalsData()), gl.STATIC_DRAW);
 79     gl.vertexAttribPointer(3, 4, gl.FLOAT, false, 0, 0);
 80 
 81 
 82     // ELEMENTS
 83     SpriteBatch.vboElemIndices = gl.createBuffer();
 84     gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, SpriteBatch.vboElemIndices);
 85     gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(SpriteBatch.mesh.getFacesData()), gl.STATIC_DRAW);
 86 
 87     // Finised setting up VAO
 88     vao_ext.bindVertexArrayOES(null);

 92     // 	children[i].bind();
 93     // }
 94 
 95     }

 97 
 98   // COLOR
 99   gl.enableVertexAttribArray(2);
100   this.vboColor = gl.createBuffer();
101   // gl.bindBuffer(gl.ARRAY_BUFFER, this.vboColor);
102   // var color = [];
103   // for (var i = 0; i < SpriteBatch.mesh.getVerticesData().length/4; i++) {
104   // 	color = color.concat(this.material.color.getVector().toArray(());

107   // gl.vertexAttribPointer(2, 4, gl.FLOAT, false, 0, 0);

109 };

113 /**
114 * Updates the batch.
115 * @param {renderContext} renderContext The render context.
116 */
117 SpriteBatch.prototype.update = function (renderContext){
118   this.renderContext = renderContext;
119 
120 	this.material.enable();
121 
122   this.material.getShader().addMatrix(this.renderContext.getCamera().getProjectionMatrix(), "projectionMatrix");
123 	this.material.getShader().addMatrix(this.renderContext.getCamera().getViewMatrix(), "viewMatrix");
124 
125 	this.material.disable();
126 
127 };

131 /**
132 * Renders a layer of the batch.
133 * @param {Number} layer The layer to render.
134 */
135 SpriteBatch.prototype.render = function (layer){
136 
137   this.material.enable();
138 
139   vao_ext.bindVertexArrayOES(SpriteBatch.vao);
140 
141   gl.enableVertexAttribArray(0);
142 
143   // if(SpriteBatch.mesh.hasTexture())
144     gl.enableVertexAttribArray(1);
145 
146   gl.enableVertexAttribArray(3);
147 
148   var cam = this.renderContext.getCamera();
149 
150   for (var key in this.renderers) {
151 
152     var renderer = this.renderers[key];
153 
154     if(renderer.getLayer() === layer && renderer.isEnabled() && !renderer.isDestroyed()){
155 
156       var isInFrustum = cam.getFrustum().testSphere(renderer.getGameObject().getTransform().getPosition(), renderer.getRadius());
157 
158       if(isInFrustum){
159 
160         gl.enableVertexAttribArray(2);
161 
162         gl.bindBuffer(gl.ARRAY_BUFFER, this.vboColor);
163 
164       	var color = [];
165       	for (var i = 0; i < SpriteBatch.mesh.getVerticesData().length/4; i++) {
166       	   color = color.concat(renderer.getMaterial().getColor().toArray());
167       	}
168       	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(color), gl.STATIC_DRAW);
169       	gl.vertexAttribPointer(2, 4, gl.FLOAT, false, 0, 0);
170 
171         renderer.gameObject.getTransform().generateMatrix();
172         this.material.getShader().addMatrix(renderer.gameObject.getTransform().getMatrix(), "modelMatrix");
173 
174         renderer.updateMaterial(this.material);
175 
176         gl.drawElements(gl.TRIANGLES, SpriteBatch.mesh.getNumFaces()*3, gl.UNSIGNED_SHORT, 0);
177 
178         this.material.reset();
179       }
180     }
181   }
182 
183   gl.disableVertexAttribArray(0);
184 	gl.disableVertexAttribArray(1);
185 	gl.disableVertexAttribArray(2);
186 	gl.disableVertexAttribArray(3);
187 
188 	this.material.disable();
189 };
