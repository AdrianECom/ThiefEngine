1 /**
  2 * @class
  3 * @extends {Component}
  4 * @classdesc This class holds de position, rotation and scale of the game object.
  5 */
  6 var Transform = function (){
  7 	Component.call(this);
  8 
  9 	this.position = new Vector3(0,0,0);
 10 	this.rotation = new Vector3(0,0,0);
 11 	this.scale = new Vector3(1,1,1);
 12 
 13 	this.right = new Vector3(1,0,0);
 14 	this.up = new Vector3(0,1,0);
 15 	this.forward = new Vector3(0,0,1);
 16 
 17   this.matrix = Matrix4.identity();
 18   this.translationMatrix = Matrix4.identity();
 19   this.rotationMatrix = Matrix4.identity();
 20   this.scaleMatrix = Matrix4.identity();
 21 
 22 	this.target = this.forward.cpy();
 23 
 24 	this.dirty = true;
 25 };
 26 
 27 Transform.prototype = new Component();
 28 Transform.prototype.constructor = Transform;

 52 /**
 53 * Generates the local space model matrix.
 54 */
 55 Transform.prototype.generateLocalSpaceMatrix = function (){
 56 
 57 	// this.translationMatrix = Matrix4.identity();
 58 	this.rotationMatrix = Matrix4.identity();
 59 	// this.scaleMatrix = Matrix4.identity();
 60 
 61 	this.scaleMatrix = Matrix4.scale(this.scale);
 62 	this.translationMatrix = Matrix4.translation(this.position);
 63 
 64 	if(this.rotation.x !== 0)
 65 		this.rotationMatrix = Matrix4.mulMM(this.rotationMatrix, Matrix4.rotation(new Vector3(this.rotation.x, 0, 0)));
 66 	if(this.rotation.y !== 0)
 67 		this.rotationMatrix = Matrix4.mulMM(this.rotationMatrix, Matrix4.rotation(new Vector3(0, this.rotation.y, 0)));
 68 	if(this.rotation.z !== 0)
 69 		this.rotationMatrix = Matrix4.mulMM(this.rotationMatrix, Matrix4.rotation(new Vector3(0, 0, this.rotation.z)));
 70 
 71 	this.matrix = Matrix4.mulMM(Matrix4.mulMM(this.scaleMatrix, this.translationMatrix),this.rotationMatrix);
 72 
 73 
 74 
 75 };

 79 /**
 80 * Initializes the model matrix.
 81 */
 82 Transform.prototype.initMatrix = function (){
 83 
 84 	this.generateLocalSpaceMatrix();
 85 
 86 	var parent = this.getParent();
 87 
 88 	if(parent !== null)
 89 		this.matrix = Matrix4.mulMM(this.matrix,parent.getMatrix());
 90 
 91 	var children = this.getChildren();
 92 
 93 	for (var c of children){
 94 		c.initMatrix();
 95 	}
 96 };

111 /**
112 * Generates the model matrix.
113 */
114 Transform.prototype.generateMatrix = function (){
115 
116 	// If this transform is not static
117 	if(! this.isStatic()){
118 
119 		var parent = this.getParent();
120 
121 		// If this transform has parent -> generate local + parent matrix
122 		if(parent !== null && parent.isDirty()){
123 			this.generateLocalSpaceMatrix();
124 			this.matrix = Matrix4.mulMM(parent.getMatrix(),this.matrix);
125 
126 			// console.log(this.position.cpy().add(parent.position));
127 			// console.log(this.position);
128 
129 		// Else, generate only local matrix
130 		}else if(this.dirty)
131 			this.generateLocalSpaceMatrix();
132 	}
133 
134 	// update children
135 	for (var c of this.getChildren()){
136 		c.generateMatrix();
137 	}
138 
139 
140 
141 	this.dirty = false; // clear dirty flag

210 /**
211 * Returns the Position.
212 * @returns {Vector3} The Position.
213 */
214 Transform.prototype.getPosition = function (){
215 
216 	var parent = this.getParent();
217 	if(parent !== null){
218 
219 		this.position.w = 1;
220 
221 		var worldPos = Matrix4.mulMV(Matrix4.mulMM(this.translationMatrix,Matrix4.mulMM(parent.translationMatrix,parent.rotationMatrix)),this.position);
222 		return worldPos;
223 	}
224 
225 	return this.position;
226 };

258 		this.setPosition(this.position.cpy().add(vec));

279 Transform.prototype.lookAt = function (vec){
280 
281 	this.dirty = true;
282 
283 	this.target = vec.cpy();
284 
285 	var pos = this.getPosition().cpy();
286 
287 	this.forward = this.target.cpy().sub(pos).nor();
288 	// this.forward = pos.sub(this.target).nor();
289 	// console.log(this.forward);
290 
291 	var yAxis = new Vector3(0,1,0);
292 	this.right = yAxis.cross(this.forward).nor();
293 	this.up = this.forward.cpy().cross(this.right);
294 
295 };
