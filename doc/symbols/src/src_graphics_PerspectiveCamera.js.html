1 /**
  2 * @class
  3 * @extends {Camera}
  4 * @classdesc This class represents an Perspective Projection camera.
  5 * @param {Number} near The distance of the near plane.
  6 * @param {Number} far The distance of the far plane.
  7 * @param {Number} aspect The aspect ratio.
  8 * @param {Number} fov The Field of View, in degrees.
  9 */
 10 var PerspectiveCamera = function (near, far, aspect, fov){
 11 	Camera.call(this,Matrix4.perspective(near, far, aspect, fov));
 12 
 13 	this.near = near;
 14 	this.far = far;
 15 	this.aspect = aspect;
 16 	this.fov = fov;
 17 };
 18 
 19 PerspectiveCamera.prototype = new Camera();
 20 PerspectiveCamera.prototype.constructor = PerspectiveCamera;
