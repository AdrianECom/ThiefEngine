1 /**
  2 * @class
  3 * @classdesc This class represents a color.
  4 * @param {Number} r The red component.
  5 * @param {Number} g The green component.
  6 * @param {Number} b The blue component.
  7 * @param {Number} a The alpha component.
  8 */
  9 var Color = function (r,g,b,a) {
 10   // this.vec = new Vector4(r,g,b,a);
 11 
 12   this.r = r;
 13   this.g = g;
 14   this.b = b;
 15   this.a = a;
 16 };
 17 
 18 // PRE-DEFINED COLORS
 19 Color.NONE = new Color(0.0, 0.0, 0.0, 0.0);
 20 Color.RED = new Color(1.0, 0.0, 0.0, 1.0);
 21 Color.GREEN = new Color(0.0, 1.0, 0.0, 1.0);
 22 Color.BLUE = new Color(0.0, 0.0, 1.0, 1.0);
 23 Color.CYAN = new Color(0.0, 1.0, 1.0, 1.0);
 24 Color.YELLOW = new Color(1.0, 1.0, 0.0, 1.0);

 28 /**
 29 * Returns a random color.
 30 * @returns {Color} The color.
 31 */
 32 Color.random = function () {
 33   var r = Math.random();
 34   var g = Math.random();
 35   var b = Math.random();
 36   // var a = Math.Random();
 37 
 38   return new Color(r,g,b,1.0);
 39 };
