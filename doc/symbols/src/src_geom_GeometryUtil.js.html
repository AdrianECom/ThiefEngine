 11 /**
 12 * Returns if the rectangle contains the point.
 13 * @param {Vector4} leftTopVertex The leftTopVertex of the rectangle.
 14 * @param {Number} width The width of the rectangle.
 15 * @param {Number} height The height of the rectangle.
 16 * @param {Number} point The point.
 17 * @returns {Boolean} True if the rectangle contains the point.
 18 */
 19 GeometryUtil.testRectanglePoint = function(leftTopVertex,width,height,point, eps){
 20 	return (leftTopVertex.x-eps < point.x && leftTopVertex.y+eps > point.y &&
 21 	leftTopVertex.x + width +eps > point.x && leftTopVertex.y - height -eps < point.y);
 22 };

 25 /**
 26 * Returns if the spheres are colliding.
 27 * @param {Vector4} centerA The center of the A sphere.
 28 * @param {Vector4} radiusA The radius of the A sphere.
 29 * @param {Number} centerB The center of the B sphere.
 30 * @param {Number} radiusB The radius of the B sphere.
 31 * @returns {Boolean} True if the spheres are colliding.
 32 */
 33 GeometryUtil.testSphereSphere = function(centerA, centerB, radiusA, radiusB){
 34 	var distance = centerA.dst(centerB);
 35 	return (distance < radiusA+radiusB);
 36 };

 52 /**
 53 * Returns the closest point in a segment to the point p.
 54 * @param {Vector4} p The p point.
 55 * @param {Vector4} a The a point of the segment.
 56 * @param {Vector4} b The b point of the segment.
 57 * @returns {Vector4} The closest point.
 58 */
 59 GeometryUtil.closestPointInSegment = function (p,a,b){
 60 
 61 	var closest = null;
 62 
 63 	var subVector = b.cpy().sub(a);
 64 
 65   if ((subVector.x === 0) && (subVector.y === 0) && (subVector.z === 0)){
 66       // It's a point not a line segment.
 67 			// return P.sub(A).len();
 68 
 69 			return a;
 70   }
 71 
 72   // Calculate the t that minimizes the distance.
 73 	// Perpendicular Distance from a Point to a Line.
 74   // var t = ((p.x - a.x) * dx + (p.y - a.y) * dy + (p.z - a.z) * dz) /
 75   //     (dx * dx + dy * dy + dz * dz);
 76 
 77 	var t = p.cpy().sub(a).dot(subVector)/(subVector.dot(subVector));
 78 
 79   // See if this represents one of the segment's
 80   // end points or a point in the middle.
 81   if (t < 0){
 82 		closest = a;
 83   }else if (t > 1){
 84 		closest = b;
 85   }else{
 86     closest = a.cpy().add(subVector.mulScl(t));
 87   }
 88 
 89 	return closest;
 90 };

 94 /**
 95 * Returns the squared distance between a point and a segment.
 96 * @param {Vector4} p The p point.
 97 * @param {Vector4} a The a point of the segment.
 98 * @param {Vector4} b The b point of the segment.
 99 * @returns {Number} The squared distance.
100 */
101 GeometryUtil.sqDistanceToSegment = function (p,a,b){
102 
103 	var closest = GeometryUtil.closestPointInSegment(p,a,b);
104 
105 	var subVector = p.cpy().sub(closest);
106 
107   // return (dx * dx + dy * dy + dz * dz);
108 	return (subVector.dot(subVector));
109 
110 };

117 * @param {Vector4} a The a point of the segment.

119 * @returns {Number} The distance.

121 GeometryUtil.distanceToSegment = function (p,a,b){
122 	return Math.sqrt(GeometryUtil.sqDistanceToSegment(p,a,b));

127 /**
128 * Returns the closest point in a sphere to the point p.
129 * @param {Vector4} p The p point.
130 * @param {Vector4} center The center of the sphere.
131 * @param {Number} b The radius of the sphere.
132 * @returns {Vector4} The closest point.
133 */
134 GeometryUtil.closestPointInSphere = function (p,center,radius){
135 	var pToCenter = p.cpy().sub(center);
136 
137 	var radiusVector = pToCenter.nor().mulScl(radius); // vector with length = radius
138 
139 	var closestPoint = center.cpy().add(radiusVector);
140 
141 	return closestPoint;
142 };
