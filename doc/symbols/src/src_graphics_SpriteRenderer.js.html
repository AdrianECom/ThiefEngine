1 /**
  2 * @class
  3 * @extends {MeshRenderer}
  4 * @classdesc This class renders a sprite, and manages the animations.
  5 */
  6 var SpriteRenderer = function (){
  7   MeshRenderer.call(this);
  8 
  9   this.setMesh(new RectangleMesh());
 10 
 11   // Animations
 12   this.animations = [];
 13   this.animation = null;
 14 
 15 };
 16 
 17 SpriteRenderer.prototype = new MeshRenderer();
 18 SpriteRenderer.prototype.constructor = SpriteRenderer;
 19 
 20 //----------------------------------------------------------------------
 21 
 22 /**
 23 * Sets the animation, by name.
 24 * @param {String} name The name.
 25 */
 26 SpriteRenderer.prototype.setAnimation = function (name){
 27 	this.animation = this.animations[name];
 28 };
 29 
 30 //----------------------------------------------------------------------
 31 
 32 /**
 33 * Adds an animation, by name.
 34 * @param {String} name The name.
 35 * @param {Animation} animation The animation.
 36 */
 37 SpriteRenderer.prototype.addAnimation = function (name, animation){
 38 	this.animations[name] = animation;
 39 };
