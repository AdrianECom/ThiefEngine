1 /**
  2 * @class
  3 * @classdesc Represents a 4x4 matrix.
  4 * @param {Vector4} row0 The 1st row.
  5 * @param {Vector4} row1 The 2nd row.
  6 * @param {Vector4} row2 The 3rd row.
  7 * @param {Vector4} row3 The 4th row.
  8 */
  9 var Matrix4 = function (row0,row1,row2,row3){
 10     this.data = new Array(16);
 11     this.transposed = null;
 12     this.setRows(row0, row1, row2, row3);
 13 
 14 
 15     // REMEMBER
 16     // WebGL uses column-major order (transposed)
 17     // a webGL matrix : [ [column 0] [column 1] [column 2] [column 3] ]
 18 
 19     // NOTE:
 20     // You can use the get(row,col) and set(row,col,value) functions
 21     // to access the data as a row-major ordered matrix.
 22 
 23     // NOTE:
 24     // Here I use, column-major order.
 25 
 26     /*
 27         ROW-MAJOR ORDER:
 28 
 29         row 0 | a b c d |
 30         row 1 | e f g h |
 31         row 2 | i j k l |
 32         row 3 | m n o p |
 33 
 34         COLUMN-MAJOR ORDER:
 35 
 36         column 0 | a e i m |
 37         column 1 | b f j n |
 38         column 2 | c g k o |
 39         column 3 | d h l p |
 40 
 41     */
 42 };

102 /**
103 * Fill the matrix with the given rows.
104 * @param {Vector4} row0 The 1st row.
105 * @param {Vector4} row1 The 2nd row.
106 * @param {Vector4} row2 The 3rd row.
107 * @param {Vector4} row3 The 4th row.
108 */
109 Matrix4.prototype.setRows = function (row0,row1,row2,row3){
110     var rows = new Array(4);
111     rows[0] = row0.toArray();
112     rows[1] = row1.toArray();
113     rows[2] = row2.toArray();
114     rows[3] = row3.toArray();
115 
116     for (var row = 0; row < 4; row++)
117         for (var col = 0; col < 4; col++)
118             this.set(row,col,rows[row][col]);
119 };

123 /**
124 * Returns the transposed matrix.
125 * @returns {Matrix4} The transposed matrix.
126 */
127 Matrix4.prototype.transpose = function () {
128 
129     if(this.transposed === null){
130 
131         this.transposed = Matrix4.zeros();
132 
133         for (var row = 0; row < 4; row++)
134           for (var col = 0; col < 4; col++)
135                 this.transposed.set(col,row,this.get(row,col));
136     }
137 
138     return this.transposed;
139 };

143 /**
144 * Multiply M1 by M2 and returns the result matrix.
145 * @param {Matrix4} M1 The 1st matrix.
146 * @param {Matrix4} M2 The 2nd matrix.
147 * @returns {Matrix4} The result matrix.
148 */
149 Matrix4.mulMM = function (M1,M2){
150 
151     // The result matrix
152     var result = this.zeros();
153 
154     for (var i = 0; i < 4; i++) {
155         for (var j = 0; j < 4; j++) {
156             for (var k = 0; k < 4; k++){
157                 result.set(i,j,result.get(i,j) + M1.get(i,k) * M2.get(k,j));
158             }
159 
160         }
161     }
162 
163     return result;
164 };

168 /**
169 * Multiply M by v and returns the result matrix.
170 * @param {Matrix4} M The matrix.
171 * @param {Vector4} v The vector.
172 * @returns {Vector4} The result vector.
173 */
174 Matrix4.mulMV = function (M,v){
175 
176     var vec = v.toArray();
177 
178     // The result vector
179     var result = new Array(4);
180 
181     for (var row = 0; row < 4; row++){
182         result[row] = 0;
183         for (var col = 0; col < 4; col++)
184             result[row] += M.get(row,col) * vec[col];
185     }
186 
187     var vectorResult = new Vector4(0,0,0,0);
188     vectorResult.fromArray(result);
189 
190     return vectorResult;
191 
192 };

239 /**
240 * Creates a rotation matrix. Each component of the vector represents an axis (x,y,z),
241 * only one component must have a non-zero value. The other two must be zero.
242 * @param {Vector3} vec The rotation.
243 * @returns {Matrix4} The result matrix.
244 */
245 Matrix4.rotation = function(vec){
246     var result = Matrix4.identity();
247 
248     var rad, cos, sin;
249 
250     if(vec.x !== 0){
251         rad = vec.x*(Math.PI/180);
252         sin = Math.sin(rad);
253         cos = Math.cos(rad);
254 
255         result = new Matrix4(
256             new Vector4(1,0,0,0),
257             new Vector4(0,cos,-sin,0),
258             new Vector4(0,sin,cos,0),
259             new Vector4(0,0,0,1));
260     }else if(vec.y !== 0){
261         rad = vec.y*(Math.PI/180);
262         sin = Math.sin(rad);
263         cos = Math.cos(rad);
264 
265         result = new Matrix4(
266             new Vector4(cos,0,sin,0),
267             new Vector4(0,1,0,0),
268             new Vector4(-sin,0,cos,0),
269             new Vector4(0,0,0,1));
270     }else if(vec.z !== 0){
271         rad = vec.z*(Math.PI/180);
272         sin = Math.sin(rad);
273         cos = Math.cos(rad);
274 
275         result = new Matrix4(
276             new Vector4(cos,-sin,0,0),
277             new Vector4(sin,cos,0,0),
278             new Vector4(0,0,1,0),
279             new Vector4(0,0,0,1));
280     }
281 
282     return result;
283 };

302 /**
303 * Creates an ortographic projection matrix.
304 * @param {Number} left The left value.
305 * @param {Number} right The right value.
306 * @param {Number} bottom The bottom value.
307 * @param {Number} top The top value.
308 * @param {Number} near The near value.
309 * @param {Number} far The far value.
310 * @returns {Matrix4} The result matrix.
311 */
312 Matrix4.ortho = function(left, right, bottom, top, near, far){
313     return new Matrix4(
314         new Vector4(2.0/(right-left),0.0,0.0,-((right+left)/(right-left))),
315         new Vector4(0.0,2.0/(top-bottom),0.0,-((top+bottom)/(top-bottom))),
316         new Vector4(0.0,0.0,(-2.0/(far-near)),-((far+near)/(far-near))),
317         new Vector4(0.0,0.0,0.0,1.0));
318 };

322 /**
323 * Creates an perspective projection matrix.
324 * @param {Number} near The near value.
325 * @param {Number} far The far value.
326 * @param {Number} aspect The aspect value.
327 * @param {Number} fov The fov value.
328 * @returns {Matrix4} The result matrix.
329 */
330 Matrix4.perspective = function(near, far, aspect, fov){
331 
332     var top = near * Math.tan((Math.PI/180)*(fov/2));
333     var bottom = -top;
334     var right = top*aspect;
335     var left = -right;
336 
337     return new Matrix4(
338         new Vector4(((2*near)/(right-left)),0,((right+left)/(right-left)),0),
339         new Vector4(0,((2*near)/(top-bottom)),((top+bottom)/(top-bottom)),0),
340         new Vector4(0,0,((far+near)/(far-near)),0),
341         new Vector4(0,0,-1,0));
342 };
