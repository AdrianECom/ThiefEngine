1 /**
  2 * @class
  3 * @classdesc This class represents a mesh.
  4 * @param {Number} numVertices The number of vertices.
  5 */
  6 var Mesh = function (numVertices) {
  7 
  8     this.numVertices = numVertices;
  9     // var size = numVertices*4;
 10 
 11     this.vertices = [];
 12     this.normals = [];
 13     this.faces = [];
 14     this.texCoord = [];
 15 
 16     this.vIndex = 0; // Vertices index
 17     this.nIndex = 0; // Normals index
 18     this.fIndex = 0; // Face index
 19     this.txIndex = 0; // TexCoord index
 20 
 21     this.max = new Vector3(0,0,0); // store the max width, height, depth;
 22 
 23 };
 24 
 25 //----------------------------------------------------------------------
 26 
 27 /**
 28 * Adds a new vertex.
 29 * @param {Vector3} vec The vertex.
 30 */
 31 Mesh.prototype.addVertex = function (vec) {
 32     this.vertices[this.vIndex] = vec.x; this.vIndex++;
 33     this.vertices[this.vIndex] = vec.y; this.vIndex++;
 34     this.vertices[this.vIndex] = vec.z; this.vIndex++;
 35     this.vertices[this.vIndex] = vec.w; this.vIndex++;
 36 
 37 
 38     // find the max width, height, depth;
 39 
 40     var x = Math.abs(vec.x)*2; // width
 41     var y = Math.abs(vec.y)*2; // height
 42     var z = Math.abs(vec.z)*2; // depth
 43 
 44     if(x > this.max.x)
 45         this.max.x = x;
 46 
 47     if(y > this.max.y)
 48         this.max.y = y;
 49 
 50     if(z > this.max.z)
 51         this.max.z = z;
 52 
 53 };
 54 
 55 //----------------------------------------------------------------------
 56 
 57 /**
 58 * Adds a new normal.
 59 * @param {Vector3} vec The normal.
 60 */
 61 Mesh.prototype.addNormal = function (vec) {
 62     this.normals[this.nIndex] = vec.x; this.nIndex++;
 63     this.normals[this.nIndex] = vec.y; this.nIndex++;
 64     this.normals[this.nIndex] = vec.z; this.nIndex++;
 65     this.normals[this.nIndex] = vec.w; this.nIndex++;
 66 };
 67 
 68 //----------------------------------------------------------------------
 69 
 70 /**
 71 * Adds a new face. A face (triangle) is composed by three indices (integers).
 72 * @param {Number} v1 The 1st index of the face.
 73 * @param {Number} v2 The 2nd index of the face.
 74 * @param {Number} v3 The 3rd index of the face.
 75 */
 76 Mesh.prototype.addFace = function (v1,v2,v3) {
 77     this.faces[this.fIndex] = v1; this.fIndex++;
 78     this.faces[this.fIndex] = v2; this.fIndex++;
 79     this.faces[this.fIndex] = v3; this.fIndex++;
 80 };
 81 
 82 //----------------------------------------------------------------------
 83 
 84 /**
 85 * Adds a new coordinate texture.
 86 * @param {Number} u The u coordinate.
 87 * @param {Number} v The v coordinate.
 88 */
 89 Mesh.prototype.addTexCoord = function (u,v) {
 90     this.texCoord[this.txIndex] = u; this.txIndex++;
 91     this.texCoord[this.txIndex] = v; this.txIndex++;
 92 };

 96 /**
 97 * Returns a vertex by index.
 98 * @param {Number} index The index of the vertex.
 99 * @returns {Vector3} The vertex.
100 */
101 Mesh.prototype.getVertex = function (index) {
102     return new Vector4(this.vertices[(index*4)],this.vertices[(index*4)+1],
103         this.vertices[(index*4)+2],this.vertices[(index*4)+3]);
104 };

118 /**
119 * Returns the number of vertices.
120 * @returns {Number} The number of vertices.
121 */
122 Mesh.prototype.getNumVertices = function () {
123     return this.vertices.length/4;
124 };

128 /**
129 * Returns the number of faces.
130 * @returns {Number} The number of faces.
131 */
132 Mesh.prototype.getNumFaces = function () {

134 };
