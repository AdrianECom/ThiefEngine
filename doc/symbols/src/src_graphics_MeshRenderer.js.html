1 /**
  2 * @class
  3 * @extends {Component}
  4 * @classdesc This class renders a Mesh.
  5 */
  6 var MeshRenderer = function (){
  7 	Component.call(this);
  8 	this.mesh = null;
  9 	this.material = null;
 10 
 11 	// texture region
 12   this.regionPosition = new Vector2(0.0,0.0);
 13   this.regionWidth = 1.0;
 14   this.regionHeight = 1.0;
 15 
 16   // alpha color
 17   this.alphaColor = new Color(-1,-1,-1,-1);
 18 
 19 	//layer
 20 	this.layer = 0;
 21 
 22 	// Animations
 23   this.animations = [];
 24   this.animation = null;
 25 };
 26 
 27 MeshRenderer.prototype = new Component();
 28 MeshRenderer.prototype.constructor = MeshRenderer;

101 /**
102 * DESCRIPTION
103 * @param {TYPE} NAME DESCRIPTION
104 * @returns {TYPE} DESCRIPTION
105 */
106 MeshRenderer.prototype.getRadius = function () {
107 
108 	var t = this.gameObject.getTransform();
109 	var scale = t.getScale();
110 
111 	var max = this.mesh.getMax();
112 
113 	var width = max.x * scale.x;
114 	var height = max.y * scale.y;
115 	var depth = max.z * scale.z;
116 
117 	return Math.sqrt(( width * width ) + (  height * height ) + (  depth * depth )) / 2.0;
118 };

213 /**
214 * DESCRIPTION
215 * @param {TYPE} NAME DESCRIPTION
216 * @returns {TYPE} DESCRIPTION
217 */
218 MeshRenderer.prototype.updateMaterial = function (material){
219   if(material.getTexture() !== null){
220 
221 
222     var shader = material.getShader();
223 
224     shader.addFloatVector(this.alphaColor.toArray(),"alphacolor");
225 
226     shader.addFloat(this.getRegionPosition().x, "regionX");
227     shader.addFloat(this.getRegionPosition().y, "regionY");
228     shader.addFloat(this.getRegionWidth(), "regionWidth");
229     shader.addFloat(this.getRegionHeight(), "regionHeight");
230 
231     if(this.animation !== null){
232       var frame = this.animation.getNextFrame();
233 
234       shader.addFloat(frame.getPosition().x, "animationX");
235     	shader.addFloat(frame.getPosition().y, "animationY");
236     	shader.addFloat(frame.getWidth(), "animationWidth");
237     	shader.addFloat(frame.getHeight(), "animationHeight");
238     }
239 
240 
241   }
242 };
