1 /**
  2 * @class
  3 * @classdesc This class represents line renderer. (NOTE: This class doesn't extend from MeshRenderer !).
  4 * @param {Shader} shader The number of vertices.
  5 * @param {Vector3} start The number of vertices.
  6 * @param {Vector3} end The number of vertices.
  7 * @param {Color} color The number of vertices.
  8 * @param {Matrix4} modelMatrix The number of vertices.
  9 */
 10 var LineRenderer = function (shader,start,end,color, modelMatrix){
 11     this.shader = shader;
 12     this.color = color;
 13 
 14     this.modelMatrix = modelMatrix;
 15 
 16     this.start = start.cpy();
 17     this.end = end.cpy();
 18 
 19     this.start.w = 1;
 20     this.end.w = 1;
 21 
 22     this.vertices = this.start.toArray().concat(this.end.toArray());
 23 
 24     this.colors = this.color.toArray().concat(this.color.toArray());
 25 
 26     this.elem = [0,1];
 27 
 28     this.vboPosition = 0;
 29   	this.vboColor = 0;
 30   	this.vboElemIndices = 0;
 31 
 32   	this.vao = 0;
 33 };
 34 
 35 //----------------------------------------------------------------------
 36 
 37 /**
 38 * Binds the line renderer. (Sends data to WebGL)
 39 */
 40 LineRenderer.prototype.bind = function () {
 41 
 42     this.vao = vao_ext.createVertexArrayOES();
 43 
 44     vao_ext.bindVertexArrayOES(this.vao);
 45 
 46     gl.enableVertexAttribArray(0);
 47 
 48   	// Create some buffers
 49   	this.vboPosition = gl.createBuffer();
 50   	gl.bindBuffer(gl.ARRAY_BUFFER, this.vboPosition);
 51   	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(this.vertices), gl.STATIC_DRAW); // TODO DYNAMIC
 52   	gl.vertexAttribPointer(0, 4, gl.FLOAT, false, 0, 0);
 53 
 54   	gl.enableVertexAttribArray(1);
 55 
 56     this.vboColor = gl.createBuffer();
 57   	gl.bindBuffer(gl.ARRAY_BUFFER, this.vboColor);
 58   	gl.bufferData(gl.ARRAY_BUFFER, new Float32Array(this.colors), gl.STATIC_DRAW); // TODO DYNAMIC
 59   	gl.vertexAttribPointer(1, 4, gl.FLOAT, false, 0, 0);
 60 
 61 
 62     this.vboElemIndices = gl.createBuffer();
 63   	gl.bindBuffer(gl.ELEMENT_ARRAY_BUFFER, this.vboElemIndices);
 64   	gl.bufferData(gl.ELEMENT_ARRAY_BUFFER, new Uint16Array(this.elem), gl.STATIC_DRAW);
 65 
 66     vao_ext.bindVertexArrayOES(null);
 67 };
 68 
 69 //----------------------------------------------------------------------
 70 
 71 
 72 /**
 73 * Renders the line.
 74 */
 75 LineRenderer.prototype.render = function () {
 76     this.shader.enable();
 77 
 78     this.shader.addMatrix(this.modelMatrix, "modelMatrix");
 79 
 80   	vao_ext.bindVertexArrayOES(this.vao);
 81 
 82   	gl.enableVertexAttribArray(0);
 83   	gl.enableVertexAttribArray(1);
 84 
 85   	gl.drawElements(gl.LINES, 2, gl.UNSIGNED_SHORT, 0);
 86 
 87     this.shader.disable();
 88 
 89   	gl.disableVertexAttribArray(0);
 90   	gl.disableVertexAttribArray(1);
 91 
 92 };
