  2 /**
  3 * @class
  4 * @classdesc This class offers line rendering.
  5 */
  6 var DebugRenderer = function (){
  7 };
  8 
  9 DebugRenderer.shader = null;
 10 DebugRenderer.modelMatrix = Matrix4.identity();
 11 
 12 DebugRenderer.list = [];
 13 
 14 //----------------------------------------------------------------------
 15 
 16 /**
 17 * Returns a list of renderers.
 18 * @returns {Array} The renderers.
 19 */
 20 DebugRenderer.getRenderers = function () {
 21     return DebugRenderer.list;
 22 };
 23 
 24 //----------------------------------------------------------------------
 25 
 26 /**
 27 * Clears the list of renderers.
 28 */
 29 DebugRenderer.clear = function () {
 30     DebugRenderer.list = [];
 31 };
 32 
 33 //----------------------------------------------------------------------
 34 
 35 /**
 36 * Inits the debug renderer.
 37 */
 38 DebugRenderer.init = function() {
 39     DebugRenderer.shader = Shader.create('debug');
 40 
 41     DebugRenderer.shader.compile();
 42 };
 43 
 44 //----------------------------------------------------------------------
 45 
 46 /**
 47 * Sets the transformation matrix. If the matrix is null, the transformation
 48 * matrix is setted to Identity.
 49 * @param {Matrix4} matrix The matrix.
 50 */
 51 DebugRenderer.setTransformationMatrix = function (matrix) {
 52     if(matrix === null)
 53         DebugRenderer.modelMatrix = Matrix4.identity();
 54     else
 55         DebugRenderer.modelMatrix = matrix;
 56 };
 57 
 58 //----------------------------------------------------------------------
 59 
 60 /**
 61 * Updates the debug renderer.
 62 * @param {RenderContext} renderContext The render context.
 63 */
 64 DebugRenderer.update = function (renderContext) {
 65 	DebugRenderer.shader.enable();
 66 
 67 	DebugRenderer.shader.addMatrix(renderContext.getCamera().getProjectionMatrix(), "projectionMatrix");
 68 	DebugRenderer.shader.addMatrix(renderContext.getCamera().getViewMatrix(), "viewMatrix");
 69 
 70 	DebugRenderer.shader.disable();
 71 };
 72 //----------------------------------------------------------------------
 73 
 74 /**
 75 * Renders the debug renderer.
 76 */
 77 DebugRenderer.render = function () {
 78   for (var lineRenderer of this.list){
 79     lineRenderer.bind();
 80     lineRenderer.render();
 81   }
 82 
 83 };
 84 
 85 //----------------------------------------------------------------------
 86 
 87 /**
 88 * Draws a line.
 89 * @param {Vector3} start The start.
 90 * @param {Vector3} end The end.
 91 * @param {Color} color The color.
 92 */
 93 DebugRenderer.drawLine = function (start,end,color){
 94     DebugRenderer.list.push(new LineRenderer(DebugRenderer.shader,start,end,color,this.modelMatrix.cpy()));
 95 };
