1 /**
  2 * @class
  3 * @extends {BaseObject}
  4 * @classdesc Represents a component of a game object. A component provides specific functionality to the game object.
  5 */
  6 var Component = function (){
  7     BaseObject.call(this);
  8     this.gameObject = null;
  9     this.enabled = true;
 10     this.destroyed = false;
 11 };
 12 
 13 Component.prototype = new BaseObject();
 14 Component.prototype.constructor = Component;
 15 
 16 //----------------------------------------------------------------------
 17 
 18 /**
 19 * Returns the game object.
 20 * @returns {GameObject} The game object.
 21 */
 22 Component.prototype.getGameObject = function (){
 23 	return this.gameObject;
 24 };
 25 
 26 //----------------------------------------------------------------------
 27 
 28 /**
 29 * Sets the game object.
 30 * @param {GameObject} gameObject The Game object.
 31 */
 32 Component.prototype.setGameObject = function (gameObject){
 33 	this.gameObject=gameObject;
 34 };
 35 
 36 //----------------------------------------------------------------------
 37 
 38 /**
 39 * Enables the component.
 40 */
 41 Component.prototype.enable = function (){
 42 	this.enabled=true;
 43 };
 44 
 45 //----------------------------------------------------------------------
 46 
 47 /**
 48 * Disables the component.
 49 */
 50 Component.prototype.disable = function (){
 51 	this.enabled=false;
 52 };
 53 
 54 //----------------------------------------------------------------------
 55 
 56 /**
 57 * Returns true if component is enabled.
 58 * @returns {Boolean} True if component is enabled.
 59 */
 60 Component.prototype.isEnabled = function (){
 61 	return this.enabled;
 62 };
 63 
 64 //----------------------------------------------------------------------
 65 
 66 /**
 67 * Returns the component's parent. The componet's parent is the component ,with the same class of
 68 * this component, in the game object's parent.
 69 * @returns {Component} The component's parent.
 70 */
 71 Component.prototype.getParent = function(){
 72     var parent = this.gameObject.getParent();
 73 
 74     if(parent !== null)
 75       return parent.getComponent(this.constructor);
 76 
 77     return null;
 78 
 79 };
 80 
 81 //----------------------------------------------------------------------
 82 
 83 /**
 84 * Returns the component's children. Returns all the components, with the same class of this component,
 85 * in the game component's children.
 86 * @returns {Component} The component's children.
 87 */
 88 Component.prototype.getChildren = function (){
 89     return this.gameObject.getComponentsInChildren(this.constructor);
 90 };
 91 
 92 //----------------------------------------------------------------------
 93 
 94 /**
 95 * Destroys the component.
 96 */
 97 Component.prototype.destroy = function (){
 98   // this.gameObject = null;
 99   this.destroyed = true;
100 };
101 
102 //----------------------------------------------------------------------
103 
104 /**
105 * Returns true if the component is destroyed.
106 * @returns {Boolean} True if the component is destroyed.
107 */
108 Component.prototype.isDestroyed = function (){
109 	return this.destroyed;
110 };
