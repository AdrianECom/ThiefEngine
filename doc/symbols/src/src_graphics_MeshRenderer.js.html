1 /**
  2 * @class
  3 * @extends {Component}
  4 * @classdesc This class renders a Mesh.
  5 */
  6 var MeshRenderer = function (){
  7 	Component.call(this);
  8 	this.mesh = null;
  9 	this.material = null;
 10 
 11 	// texture region
 12   this.regionPosition = new Vector2(0.0,0.0);
 13   this.regionWidth = 1.0;
 14   this.regionHeight = 1.0;
 15 
 16   // alpha color
 17   this.alphaColor = new Color(-1,-1,-1,-1);
 18 
 19 	//layer
 20 	this.layer = 0;
 21 };
 22 
 23 MeshRenderer.prototype = new Component();
 24 MeshRenderer.prototype.constructor = MeshRenderer;

 97 /**
 98 * DESCRIPTION
 99 * @param {TYPE} NAME DESCRIPTION
100 * @returns {TYPE} DESCRIPTION
101 */
102 MeshRenderer.prototype.getRadius = function () {
103 
104 	var t = this.gameObject.getTransform();
105 	var scale = t.getScale();
106 
107 	var max = this.mesh.getMax();
108 
109 	var width = max.x * scale.x;
110 	var height = max.y * scale.y;
111 	var depth = max.z * scale.z;
112 
113 	return Math.sqrt(( width * width ) + (  height * height ) + (  depth * depth )) / 2.0;
114 };

186 /**
187 * DESCRIPTION
188 * @param {TYPE} NAME DESCRIPTION
189 * @returns {TYPE} DESCRIPTION
190 */
191 MeshRenderer.prototype.updateMaterial = function (material){
192   if(material.getTexture() !== null){
193 
194 
195     var shader = material.getShader();
196 
197     shader.addFloatVector(this.alphaColor.toArray(),"alphacolor");
198 
199     shader.addFloat(this.getRegionPosition().x, "regionX");
200     shader.addFloat(this.getRegionPosition().y, "regionY");
201     shader.addFloat(this.getRegionWidth(), "regionWidth");
202     shader.addFloat(this.getRegionHeight(), "regionHeight");
203 
204     if(this.animation !== null){
205       var frame = this.animation.getNextFrame();
206 
207       shader.addFloat(frame.getPosition().x, "animationX");
208     	shader.addFloat(frame.getPosition().y, "animationY");
209     	shader.addFloat(frame.getWidth(), "animationWidth");
210     	shader.addFloat(frame.getHeight(), "animationHeight");
211     }
212 
213 
214   }
215 };
