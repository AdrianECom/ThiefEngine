  3 /**
  4 * @class
  5 * @classdesc This class manage the graphics of the application.
  6 */
  7 var RenderEngine = function (){
  8 
  9   this.color = new Color(0,0,0,1);
 10   this.numLayers = 0;
 11 
 12   var canvas = document.getElementById("glcanvas");
 13 
 14   // ########################################
 15   // PATCH
 16   // http://webglfundamentals.org/webgl/lessons/webgl-resizing-the-canvas.html
 17   // ########################################
 18   var realToCSSPixels = window.devicePixelRatio || 1; // FOR HD RETINA SCREEN
 19 
 20   // Lookup the size the browser is displaying the canvas in CSS pixels
 21   // and compute a size needed to make our drawingbuffer match it in
 22   // device pixels.
 23   var displayWidth  = Math.floor(canvas.clientWidth  * realToCSSPixels);
 24   var displayHeight = Math.floor(canvas.clientHeight * realToCSSPixels);
 25 
 26   canvas.width = displayWidth;
 27   canvas.height = displayHeight;
 28 
 29   // ########################################
 30 
 31   try {
 32     // Try to grab the standard context. If it fails, fallback to experimental.
 33     gl = canvas.getContext("webgl") || canvas.getContext("experimental-webgl");
 34 
 35   } catch(e) {
 36     alert("Unable to initialize WebGL. Your browser may not support it.");
 37   }
 38 
 39   // If we don't have a GL context.
 40   if (!gl) {
 41     alert("Unable to initialize WebGL. Your browser may not support it.");
 42     gl = null;
 43   }
 44 
 45   vao_ext = gl.getExtension ("OES_vertex_array_object");
 46   instance_ext = gl.getExtension("ANGLE_instanced_arrays");
 47 
 48   if (!vao_ext) {
 49     alert("ERROR: Your browser does not support WebGL 'VAO' extension");
 50   }
 51 
 52   if (!instance_ext) {
 53     alert("ERROR: Your browser does not support WebGL 'ANGLE Instanced Rendering' extension");
 54   }
 55 
 56   // Only continue if WebGL is available and working
 57 
 58   if (gl) {
 59 
 60     gl.viewport(0, 0, gl.drawingBufferWidth, gl.drawingBufferHeight);
 61     gl.clearColor(this.color.r, this.color.g, this.color.b, this.color.a); // Set clear color.
 62 
 63     gl.enable(gl.DEPTH_TEST); // Enable depth testing
 64     gl.depthFunc(gl.LEQUAL); // Near things obscure far things
 65     gl.enable(gl.CULL_FACE); // BACK by default
 66     gl.cullFace(gl.BACK);
 67 
 68     gl.blendFunc(gl.SRC_ALPHA, gl.ONE_MINUS_SRC_ALPHA);
 69     // gl.enable(gl.BLEND);
 70 
 71     gl.clear(gl.COLOR_BUFFER_BIT|gl.DEPTH_BUFFER_BIT);      // Clear the color and the depth buffer.
 72   }
 73 
 74     this.clear();
 75 
 76     DebugRenderer.init();
 77 };

108 /**
109 * Adds a list of renderers.
110 * @param {Array} renderers The renderers.
111 */
112 RenderEngine.prototype.addRenderers = function (renderers){
113 
114   for (var renderer of renderers) {
115 
116     this.numLayers = Math.max(this.numLayers,renderer.getLayer()); // Always check the max number of layers.
117 
118 
119     // BATCHING BY TEXTURE
120 
121     var tex = renderer.getMaterial().getTexture();
122 
123     // if renderer has a texture.
124     if(tex !== null){
125 
126       var texName = tex.getName();
127 
128       // if the batch exists.
129       if((texName in this.textureBatches)){
130           this.textureBatches[texName].add(renderer);
131       }else {
132           this.textureBatches[texName] = new SpriteBatch(renderer.getMaterial());
133           this.textureBatches[texName].add(renderer);
134       }
135 
136     }else // if renderer has not a texture.
137       this.noTextureBatch.add(renderer);
138 
139   }
140 };

166 /**
167 * Updates the render engine.
168 */
169 RenderEngine.prototype.update = function (){
170 
171   // RE-BUILD frustum
172   this.renderContext.getCamera().getFrustum().build();
173 
174   for (var i in this.textureBatches)
175     this.textureBatches[i].update(this.renderContext);
176 
177   this.noTextureBatch.update(this.renderContext);
178 
179   DebugRenderer.update(this.renderContext);
180 
181 };

185 /**
186 * Binds all the batches.
187 */
188 RenderEngine.prototype.bind = function (){
189 
190   for (var i in this.textureBatches)
191     this.textureBatches[i].bind();
192 
193   this.noTextureBatch.bind();
194 };

198 /**
199 * Renders all the batches and the debug renderer.
200 */
201 RenderEngine.prototype.render = function (){
202 
203   // TODO: culling ????
204 
205   for (var i = 0; i <= this.numLayers; i++) {
206     for (var j in this.textureBatches)
207       this.textureBatches[j].render(i);
208       this.noTextureBatch.render(i);
209   }
210 
211   DebugRenderer.render();
212   DebugRenderer.clear();
213 
214 };
