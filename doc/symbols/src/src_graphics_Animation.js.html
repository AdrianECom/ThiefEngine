1 /**
  2 * @class
  3 * @classdesc This class represents a texture animation.
  4 */
  5 var Animation = function () {
  6   this.frames = [];
  7   this.currentFrame = 0;
  8   this.speed = 1;
  9   this.lastTime = 0;
 10 
 11 };
 12 
 13 //----------------------------------------------------------------------
 14 
 15 /**
 16 * Creates an animation.
 17 * @param {Number} frameCount The animation's number of frames.
 18 * @param {Boolean} horizontal True if horizontal (left to right), False if vertical (down to up).
 19 * @param {Boolean} reverse True to use the reverse direction.
 20 * @param {Vector2} startPosition The start position (bottom-left corner of the frame) in texture coordinates (see OpenGL texture coordinates).
 21 * @param {Number} width The frame width.
 22 * @param {Number} height The frame height.
 23 * @param {Number} speed The animation's speed.
 24 * @returns {Animation} The animation.
 25 */
 26 Animation.create = function (frameCount, horizontal, reverse, startPosition, width, height, speed) {
 27 
 28     // TODO: check if coordinates are > 1 or < 0 !!!!!
 29 
 30     var animation = new Animation();
 31     animation.setSpeed(speed);
 32 
 33     var horizontalDir = 0;
 34     var verticalDir = 0;
 35 
 36     if(horizontal)
 37       horizontalDir = 1;
 38     else
 39       verticalDir = 1;
 40 
 41     if(reverse){
 42       horizontalDir *= -1;
 43       verticalDir *= -1;
 44     }
 45 
 46     var start = 0;
 47     var end = frameCount;
 48     var delta = 1;
 49 
 50     if(reverse){
 51       start = frameCount-1;
 52       end = 0;
 53       delta = -1;
 54     }
 55 
 56     for (var i = start; i !== end; i+=delta) {
 57 
 58       var pos = new Vector2(0,0);
 59       pos.add(startPosition);
 60 
 61       if(horizontalDir !== 0)
 62         pos.x += i*width;
 63 
 64       if(verticalDir !== 0)
 65         pos.y += i*height;
 66 
 67       var frame = new AnimationFrame();
 68       frame.set(pos,width,height);
 69       animation.addFrame(frame);
 70 
 71     }
 72 
 73     return animation;
 74 };

 78 /**
 79 * Sets the animation's speed.
 80 * @param {Number} speed The animation's speed.
 81 */
 82 Animation.prototype.setSpeed = function (speed) {
 83     this.speed = speed;
 84 };
 85

 88 /**
 89 * Adds a frame to the animation.
 90 * @param {AnimationFrame} frame The frame.
 91 */
 92 Animation.prototype.addFrame = function (frame) {
 93     this.frames.push(frame);
 94 };

 98 /**

100 * @returns {Number} The number of frames.
101 */
102 Animation.prototype.getNumberOfFrames = function () {
103     return this.frames.length;
104 };
105

108 /**
109 * Returns the current frame number.
110 * @returns {Number} The current frame number.
111 */
112 Animation.prototype.getCurrentFrameNumber = function () {
113     return this.currentFrame;
114 };

116 //----------------------------------------------------------------------
117 
118 /**
119 * Returns the next frame.
120 * @returns {AnimationFrame} The next frame.
121 */
122 Animation.prototype.getNextFrame = function () {
123 
124     // this.speed -> frame/second.
125     // var time -> time of one frame.
126 
127     var time = (1.0/(this.speed)); // in seconds !
128     var now = Time.now(); // in seconds !
129 
130     // if delta time is greater than 'one frame time'
131     // then -> change to the next frame.
132     if((now-this.lastTime) >= time){
133         this.lastTime = now;
134         this.currentFrame = (this.currentFrame + 1)%this.frames.length;
135     }
136 
137     return this.frames[this.currentFrame];
138 };
