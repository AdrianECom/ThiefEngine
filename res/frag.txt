precision mediump float;

            // Passed in from the vertex shader.
            varying vec2 v_texcoord;

            // The texture.
            uniform sampler2D uSampler;

            void main() {
              //gl_FragColor = vec4(0, 1, 0, 1);  // green
              gl_FragColor = texture2D(uSampler, v_texcoord);
            }
