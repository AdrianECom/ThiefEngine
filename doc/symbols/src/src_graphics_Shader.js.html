134 /**
135 * Adds a vertex shader.
136 * @param {String} sourceText The source.
137 */
138 Shader.prototype.addVertexShader = function (sourceText){
139 
140   var shaderID = gl.createShader(gl.VERTEX_SHADER);
141   gl.shaderSource(shaderID,sourceText);
142 
143   this.vertShaderID = shaderID;
144 };
145 
146 //----------------------------------------------------------------------
147 
148 /**
149 * Adds a fragment shader.
150 * @param {String} sourceText The source.
151 */
152 Shader.prototype.addFragmentShader = function (sourceText){
153 
154   var shaderID = gl.createShader(gl.FRAGMENT_SHADER);
155   gl.shaderSource(shaderID,sourceText);
156 
157   this.fragShaderID = shaderID;
158 };

163 * Adds an attribute. Binds an attribute to a location.
164 * @param {Number} location The location of the attribute.

167 Shader.prototype.addAttribute = function (location,name){
168   gl.bindAttribLocation(this. programID, location, name);

173 /**
174 * Sends a matrix to the shader.
175 * @param {Matrix4} matrix The matrix.
176 * @param {String} name The name.
177 */
178 Shader.prototype.addMatrix = function (matrix,name){
179   matrixLocation = gl.getUniformLocation(this.programID, name);
180   gl.uniformMatrix4fv(matrixLocation, false, new Float32Array(matrix.getData()));
181 };
182 
183 //----------------------------------------------------------------------
184 
185 /**
186 * Sends an array of matrices to the shader.
187 * @param {Array} matrixArray The array.
188 * @param {String} name The name.
189 */
190 Shader.prototype.addMatrixArray = function (matrixArray,name){
191   matrixLocation = gl.getUniformLocation(this.programID, name);
192 
193   var raw = [];
194   for (var m of matrixArray) {
195       raw = raw.concat(matrix.getData());
196   }
197 
198   gl.uniformMatrix4fv(matrixLocation, false, new Float32Array(raw));
199 };

239 /**
240 * Compiles the shader.
241 */
242 Shader.prototype.compile = function (){
243   if( ! this.isCompiled()){
244     gl.compileShader(this.fragShaderID);
245 
246     var compiled = gl.getShaderParameter(this.fragShaderID, gl.COMPILE_STATUS);
247     //    console.log('Shader compiled successfully: ' + compiled); // TODO uncomment console.log()
248     var compilationLog = gl.getShaderInfoLog(this.fragShaderID);
249     //    console.log('Shader compiler log: ' + compilationLog);
250 
251     gl.compileShader(this.vertShaderID);
252 
253     compiled = gl.getShaderParameter(this.vertShaderID, gl.COMPILE_STATUS);
254     //    console.log('Shader compiled successfully: ' + compiled);
255     compilationLog = gl.getShaderInfoLog(this.vertShaderID);
256     //    console.log('Shader compiler log: ' + compilationLog);
257 
258     gl.attachShader(this.programID,this.fragShaderID);
259     gl.attachShader(this.programID,this.vertShaderID);
260     gl.linkProgram(this.programID);
261 
262     this.compiled = true;
263   }
264 };

286 /**
287 * Creates a shader from a type.
288 * @param {String} typeStr The type.
289 */
290 Shader.create = function(typeStr){
291 
292   var shader = null;
293 
294   if(typeStr === 'default'){
295 
296     if(Shader.defaultShader === null){
297 
298       Shader.defaultShader = new Shader();
299 
300       Shader.defaultShader.addVertexShader(Shader.vs);
301       Shader.defaultShader.addFragmentShader(Shader.fs);
302       Shader.defaultShader.addAttribute(0, "position");
303 
304       //REMEMBER: IF AN ATTRIBUTE IS NEVER USED getAttribLocation RETURNS -1 !!!
305       Shader.defaultShader.addAttribute(2, "color");
306       Shader.defaultShader.addAttribute(1, "texcoord");
307     }
308 
309     shader = Shader.defaultShader;
310 
311   }else if(typeStr === 'debug'){
312 
313     if(Shader.debugShader === null){
314 
315       Shader.debugShader = new Shader();
316 
317       Shader.debugShader.addVertexShader(Shader.vsDebug);
318       Shader.debugShader.addFragmentShader(Shader.fsDebug);
319       Shader.debugShader.addAttribute(0, "position");
320 
321       //REMEMBER: IF AN ATTRIBUTE IS NEVER USED getAttribLocation RETURNS -1 !!!
322       Shader.debugShader.addAttribute(1, "color");
323     }
324 
325     shader = Shader.debugShader;
326   }
327 
328   return shader;
329 };
