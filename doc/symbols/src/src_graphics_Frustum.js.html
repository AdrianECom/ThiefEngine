1 /**
  2 * @class
  3 * @classdesc This class represents a Frustum
  4 *	(see: https://en.wikipedia.org/wiki/Frustum).
  5 */
  6 var Frustum = function (camera){
  7 	this.camera = camera;
  8 	this.planes = new Array(6);
  9 
 10   for (var i = 0; i < this.planes.length; i++) {
 11       this.planes[i] = new Vector3(0,0,0); // x,y,z,w <=> A,B,C,D
 12   }
 13 
 14 	this.VPmatrix = Matrix4.identity();
 15 };
 16 
 17 //----------------------------------------------------------------------
 18 
 19 /**
 20 * Tests if sphere is touching the frustum.
 21 * @param {Vector3} center The center.
 22 * @param {Number} radius The radius.
 23 * @returns {Boolean} True if sphere is touching the frustum.
 24 */
 25 Frustum.prototype.testSphere = function (center, radius){
 26 	// console.log("TEST SPHERE");
 27 	for (var i = 0; i < this.planes.length; i++) {
 28 
 29 		var A = this.planes[i].x;
 30 		var B = this.planes[i].y;
 31 		var C = this.planes[i].z;
 32 		var D = this.planes[i].w;
 33 
 34 		// console.log(A+" "+B+" "+C+" "+D);
 35 		// console.log(-radius);
 36 		// console.log((A*center.x)+(B*center.y)+(C*center.z)+D);
 37 
 38 		if((A*center.x)+(B*center.y)+(C*center.z)+D <= -radius)
 39 			return false;
 40 	}
 41 
 42 	return true;
 43 };

 47 /**
 48 * Builds the frustum planes.
 49 */
 50 Frustum.prototype.build = function (){
 51 
 52 	var t = this.camera.getGameObject().getTransform();
 53 
 54 	if(t.isDirty()){
 55 
 56 	  var LEFT = 0;
 57 	  var RIGHT = 1;
 58 	  var BOTTOM = 2;
 59 	  var TOP = 3;
 60 	  var NEAR = 4;
 61 	  var FAR = 5;
 62 
 63 		// var camPos = t.position;
 64 
 65 	  // this.VPmatrix = Matrix4.mulMM(this.camera.getProjectionMatrix(),this.camera.getViewMatrix());
 66 	  this.VPmatrix = Matrix4.mulMM(this.camera.getViewMatrix().transpose(),this.camera.getProjectionMatrix().transpose());
 67 		this.VPmatrix = this.VPmatrix.transpose();
 68 
 69 		// this.VPmatrix = Matrix4.mulMM(Matrix4.translation(new Vector3(-3,0,0)),this.VPmatrix)
 70 
 71 		var d = new Array(6);
 72 
 73 		/*
 74 		* LEFT
 75 		*/
 76 	  this.planes[LEFT].x = this.VPmatrix.get(0,0) + this.VPmatrix.get(3,0);
 77 	  this.planes[LEFT].y = this.VPmatrix.get(0,1) + this.VPmatrix.get(3,1);
 78 	  this.planes[LEFT].z = this.VPmatrix.get(0,2) + this.VPmatrix.get(3,2);
 79 		d[LEFT] =  this.VPmatrix.get(0,3) + this.VPmatrix.get(3,3);
 80 
 81 		/*
 82 		* RIGHT
 83 		*/
 84 	  this.planes[RIGHT].x = -this.VPmatrix.get(0,0) + this.VPmatrix.get(3,0);
 85 	  this.planes[RIGHT].y = -this.VPmatrix.get(0,1) + this.VPmatrix.get(3,1);
 86 	  this.planes[RIGHT].z = -this.VPmatrix.get(0,2) + this.VPmatrix.get(3,2);
 87 		d[RIGHT] =  -this.VPmatrix.get(0,3) + this.VPmatrix.get(3,3);
 88 
 89 		/*
 90 		* BOTTOM
 91 		*/
 92 	  this.planes[BOTTOM].x = this.VPmatrix.get(1,0) + this.VPmatrix.get(3,0);
 93 	  this.planes[BOTTOM].y = this.VPmatrix.get(1,1) + this.VPmatrix.get(3,1);
 94 	  this.planes[BOTTOM].z = this.VPmatrix.get(1,2) + this.VPmatrix.get(3,2);
 95 	  d[BOTTOM] =  this.VPmatrix.get(1,3) + this.VPmatrix.get(3,3);
 96 
 97 		/*
 98 		* TOP
 99 		*/
100 	  this.planes[TOP].x = -this.VPmatrix.get(1,0) + this.VPmatrix.get(3,0);
101 	  this.planes[TOP].y = -this.VPmatrix.get(1,1) + this.VPmatrix.get(3,1);
102 	  this.planes[TOP].z = -this.VPmatrix.get(1,2) + this.VPmatrix.get(3,2);
103 	  d[TOP] =  -this.VPmatrix.get(1,3) + this.VPmatrix.get(3,3);
104 
105 		/*
106 		* NEAR
107 		*/
108 	  this.planes[NEAR].x = this.VPmatrix.get(2,0) + this.VPmatrix.get(3,0);
109 	  this.planes[NEAR].y = this.VPmatrix.get(2,1) + this.VPmatrix.get(3,1);
110 	  this.planes[NEAR].z = this.VPmatrix.get(2,2) + this.VPmatrix.get(3,2);
111 	  d[NEAR] =  this.VPmatrix.get(2,3) + this.VPmatrix.get(3,3);
112 
113 		/*
114 		* FAR
115 		*/
116 	  this.planes[FAR].x = -this.VPmatrix.get(2,0) + this.VPmatrix.get(3,0);
117 	  this.planes[FAR].y = -this.VPmatrix.get(2,1) + this.VPmatrix.get(3,1);
118 	  this.planes[FAR].z = -this.VPmatrix.get(2,2) + this.VPmatrix.get(3,2);
119 	  d[FAR] =  -this.VPmatrix.get(2,3) + this.VPmatrix.get(3,3);
120 
121 
122     for (var i = 0; i < this.planes.length; i++) {
123 
124 			if(this.planes[i].len() > 0){
125 
126 				d[i] = d[i]/this.planes[i].len();
127 
128 				this.planes[i] = this.planes[i].nor();
129 
130 				this.planes[i].w = d[i];
131 			}
132     }
133 	}
134 };
