 46 /**
 47 * Sets the current scene as the scene named as 'name'.
 48 * @param {String} name The scene's name.
 49 */
 50 Engine.prototype.setCurrentScene = function (name){
 51 
 52   if(this.currentScene === null)
 53     this.currentScene=this.scenes[name];
 54   else if(this.currentScene.getName() !== name){
 55     this.currentScene.reset(); // reset old scene
 56     this.currentScene=this.scenes[name];
 57   }
 58 
 59 
 60 };

135 /**
136 * Updates the current scene.
137 */
138 Engine.prototype.updateScene = function(){
139 
140   var root = this.currentScene.getNewsRoot();
141 
142   var renderers = root.getComponentsInChildren(MeshRenderer);
143   var rigidBodies = root.getComponentsInChildren(RigidBody);
144   var scripts = root.getComponentsInChildren(Script);
145 
146   this.currentScene.flush();
147 
148 
149   this.renderEngine.addRenderers(renderers);
150   this.renderEngine.setRenderContext(this.currentScene.getRenderContext());
151 
152   this.physicsEngine.addBodies(rigidBodies);
153 
154   this.scriptEngine.addScripts(scripts);
155 
156   // console.log("update");
157 
158 };

162 /**
163 * Loads the current scene.
164 */
165 Engine.prototype.loadScene = function(){
166 
167   this.renderEngine.clear();
168   this.physicsEngine.clear();
169   this.scriptEngine.clear();
170 
171   this.updateScene();
172 
173   this.loaded = false;
174 
175   // console.log("load");
176 
177 };

182 /**
183 * Runs the engine.
184 */
185 Engine.prototype.run = function () {
186 
187   var step = (1/this.fps);
188 
189   window.requestAnimFrame = (function(){
190     return  window.requestAnimationFrame       ||
191             window.webkitRequestAnimationFrame ||
192             window.mozRequestAnimationFrame    ||
193             function( callback ){
194               window.setTimeout(callback, step*1000);
195             };
196   })();
197 
198   // window.requestAnimFrame = (function(){
199   //   return function( callback ){
200   //             window.setTimeout(callback, step*1000);
201   //           };
202   // })();
203 
204   var renderEngine = this.renderEngine;
205   var physicsEngine = this.physicsEngine;
206   var scriptEngine = this.scriptEngine;
207 
208   var engine = this;
209 
210   var dt = 0;
211 
212   var main = function () {
213 
214     Time.tick();
215     // dt = dt + Math.min(step,Time.deltaTime());
216     dt = dt + Time.deltaTime();
217 
218     gl.clear(gl.COLOR_BUFFER_BIT|gl.DEPTH_BUFFER_BIT);
219 
220     if(! engine.getCurrentScene().isLoaded())
221       engine.loadScene();
222 
223     if(engine.getCurrentScene().hasNewObjects())
224       engine.updateScene();
225 
226     // console.log(renderEngine.isBinded());
227     if(Loader.isDone() && !renderEngine.isBinded()){
228       engine.loaded = true;
229       renderEngine.bind();
230       // Loader.reset();
231     }
232 
233 
234 
235     if(engine.loaded){
236 
237 
238       scriptEngine.update();
239 
240       while(dt > step){
241         dt = dt - step;
242 
243         if(engine.physicsEnabled)
244           physicsEngine.update(step);
245       }
246 
247       Time.delta = dt*1000;
248       Time.deltaInSeconds = dt;
249 
250       renderEngine.update();
251       renderEngine.render();
252 
253       engine.getCurrentScene().cleanTrash();
254 
255     }
256 
257     window.requestAnimFrame(main);
258   };
259 
260   Time.init();
261 
262   window.requestAnimFrame(main);
263 
264 };
