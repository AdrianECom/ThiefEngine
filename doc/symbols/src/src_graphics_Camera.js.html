  2 /**
  3 * @class
  4 * @extends {Component}
  5 * @classdesc This class represents a camera.
  6 * @param {Matrix4} matrix The matrix. The matrix can be Perspective or Orthogonal.
  7 */
  8 var Camera = function (matrix){
  9 	Component.call(this);
 10 	this.target = null;
 11 	this.projectionMatrix = matrix;
 12 	this.frustum = new Frustum(this);
 13 	this.viewMatrix = Matrix4.identity();
 14 };
 15 
 16 Camera.prototype = new Component();
 17 Camera.prototype.constructor = Camera;

 51 /**
 52 * Returns the view matrix.
 53 * @returns {Matrix4} The viewMatrix.
 54 */
 55 Camera.prototype.getViewMatrix = function (){
 56 
 57 	var t = this.gameObject.getTransform();
 58 
 59 	if(t.isDirty()){
 60 		
 61 		var rotationMatrix = new Matrix4(
 62 	        t.right,
 63 	        t.up,
 64 	        t.forward,
 65 	        new Vector4(0,0,0,1));
 66 
 67 		var translationMatrix = Matrix4.translation(t.getPosition().cpy().mulScl(-1));
 68 
 69 		this.viewMatrix = Matrix4.mulMM(translationMatrix, rotationMatrix);
 70 	}
 71 
 72 	return this.viewMatrix;
 73 };
