 48 /**
 49 * Loads the image in async mode.
 50 * NOTE: Do not worry about the 'async' word.
 51 * The engine will wait until all the images are loaded.
 52 * @param {String} path The path.
 53 */
 54 Loader.loadImageFile = function (path){
 55 
 56     Loader.hasImageRequests = true;
 57 
 58     if((path in Loader.images)){
 59         return Loader.images[path];
 60 
 61     }else{
 62         Loader.requests++;
 63         var img = new Image();
 64 
 65         Loader.images[path] = img;
 66 
 67         img.src = path; // DOWNLOADING!
 68 
 69         img.onload = function(){
 70             Loader.loaded++;
 71             if(Loader.requests == Loader.loaded)
 72                 Loader.done = true;
 73             else
 74                 Loader.done = false;
 75         };
 76 
 77         return Loader.images[path];
 78     }
 79 
 80 };

 84 /**
 85 * Loads the texture in async mode.
 86 * NOTE: Do not worry about the 'async' word.
 87 * The engine will wait until all the images are loaded.
 88 * @param {String} path The path.
 89 */
 90 Loader.loadTexture = function (path) {
 91     if( ! (path in Loader.textures)){
 92         var img = Loader.loadImageFile(path);
 93 
 94         var texture = new Texture(path,img);
 95         Loader.textures[path] = texture;
 96     }
 97 
 98     return Loader.textures[path];
 99 };
